function doGet(e){try{var ss=SpreadsheetApp.getActiveSpreadsheet();var p=e.parameter;if(p.type==='backup'){var bs=ss.getSheetByName('Backup');if(!bs)bs=ss.insertSheet('Backup');bs.clear();bs.getRange(1,1).setValue(p.data);return ContentService.createTextOutput('Backup OK');}if(p.type==='getbackup'){var bs=ss.getSheetByName('Backup');if(!bs)return ContentService.createTextOutput('{}');var d=bs.getRange(1,1).getValue();return ContentService.createTextOutput(d).setMimeType(ContentService.MimeType.JSON);}var sn=p.type==='gst'?'GST Bills':p.type==='payment'?'Payments':'Cash Bills';var sh=ss.getSheetByName(sn);if(!sh)sh=ss.insertSheet(sn);if(sh.getLastRow()===0)sh.appendRow(['Date','Bill No','Customer','GSTIN','Items','Qty','Total','Paid','Balance','Base','CGST','SGST','Status']);sh.appendRow([p.date,p.billNo,p.dealer,p.gstin||'',p.itemsSummary,p.totalQty,p.total,p.paid||p.total,p.balance||0,p.base,p.cgst,p.sgst,p.status]);return ContentService.createTextOutput('OK');}catch(err){return ContentService.createTextOutput('Error: '+err.message);}}
